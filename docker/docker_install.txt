# 拉取源码
git clone https://github.com/Lafinger/MusicGen.git

# 复制huggingface目录到-》*\MusicGen\docker目录下

# 进入docker目录
cd MusicGen
cd docker

# 拉取基础docker镜像
docker pull pytorch/pytorch:2.1.0-cuda12.1-cudnn8-runtime

# 构建musicgen镜像 端口5000
缓存执行：docker build -f .\Dockerfile.musicgen.cuda121.models.tiny -t musicgen:cuda121-models-tiny .
无缓存执行：docker build --no-cache -f .\Dockerfile.musicgen.cuda121.models.tiny -t musicgen:cuda121-models-tiny .

# 运行musicgen镜像 端口5000
docker run -d --restart=unless-stopped --gpus all -p 5000:5000 musicgen:cuda121-models-tiny
